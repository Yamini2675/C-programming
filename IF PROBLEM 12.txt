#include<stdio.h>

int main() {
    int mark;
    scanf("%d",&mark);

    if (mark<=100 && mark>=90) {
        printf("Grade A");
    }
    else if (mark<90 &&mark>=75) {
        printf("Grade B");
    }
    else if (mark<75 &&mark>=50) {
        printf("Grade C");
    }
    else if (mark<50 &&mark>=25) {
        printf("Grade D");
    }
    else if (mark<25 &&mark>=0) {
        printf("Fail");
    }
    else {
        printf("invalid Mark");
    }
    return 0;

}
