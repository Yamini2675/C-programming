#include<stdio.h>

int palindrome(int n) {
    int temp=n;
    int rev=0;
    while (n!=0) {
        int rem=n%10;
        rev=rev*10+rem;
        n=n/10;
    }
    if (temp==rev) {
        printf("Palindrome");
    }
    else {
        printf("Not Palindrome");
    }
}

int main() {
    int n;
    scanf("%d",&n);
    palindrome(n);

    return 0;
}
