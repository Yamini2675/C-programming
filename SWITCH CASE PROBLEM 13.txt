#include<stdio.h>

int main() {
    int loan_type,credit_score;
    printf("Enter Loan Type:");
    scanf("%d",&loan_type);
    printf("Enter Credit Score:");
    scanf("%d",&credit_score);

    switch (loan_type) {
        case 1:
            if (credit_score>=700) {
                printf("Approved");
            }
            else if (credit_score<=699&&credit_score>=650) {
                printf("Manual Review");
            }
            else {
                printf("Rejected");
            }
            break;
        case 2:
            if (credit_score>=700) {
                printf("Approved");
            }
            else if (credit_score<=699&&credit_score>=650) {
                printf("Manual Review");
            }
            else {
                printf("Rejected");
            }
            break;
        default:
            printf("Invalid Loan Type");
    }
    return 0;
}
