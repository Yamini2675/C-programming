#include<stdio.h>

int armstrong(int n) {
    int org=n,temp=n;
    int rem,sum=0;
    int count=0,pow;
    while (temp!=0) {
        count++;
        temp=temp/10;
    }

    temp=n;
    while(temp!=0) {
        rem=temp%10;
        pow=1;
        for (int i=1;i<=count;i++) {
            pow=pow*rem;
        }
        sum=sum+pow;
        temp=temp/10;
    }
    return(sum==org);

}

int main() {
    int n;
    scanf("%d",&n);
    if (armstrong(n)) {
        printf("Armstrong");
    }
    else {
        printf("Not Armstrong");
    }
    return 0;
}
