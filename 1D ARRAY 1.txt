#include<stdio.h>

int main() {
    int size;
    scanf("%d",&size);
    int arr[size];
    for (int i=0;i<size;i++) {
        scanf("%d",&arr[i]);
    }
    int max=arr[0];
    int second=arr[0];

    for (int j=0;j<size;j++) {
        if (arr[j]>max) {
            second=max;
            max=arr[j];
        }
        else if (arr[j]<max) {
            if (second==max || second<arr[j]) {
                second=arr[j];
            }
        }
    }
    printf("%d",second);
    return 0;
}
