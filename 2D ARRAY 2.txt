#include<stdio.h>

int main() {
    int r,c;
    scanf("%d %d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int max1=0,min1=1;
    int max2=0,min2=1;
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
           int num=arr[i][j];
            if (num>max1) {
                max1=num;
                max2=max1;
            }
            else if (num<min1) {
                min2=min1;
                min1=num;
            }
            else if (num>max2) {
                max2=num;
            }
            else if (num<min2) {
                min2=num;
            }
        }
    }

    int pro1=max1*max2;
    int pro2=min1*min2;

    if (pro1>pro2) {
        printf("%d",pro1);
    }
    else {
        printf("%d",pro2);
    }
    return 0;
}
