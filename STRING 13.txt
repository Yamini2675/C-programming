#include<stdio.h>

int main() {
    char str[100],n;
    fgets(str,sizeof(str),stdin);
    scanf("%c",&n);
    int ind=0;

    for (int i=0;str[i]!='\0';i++) {
        if (str[i]==n) {
            ind=i;
            break;
        }
        if (str[i]!=n) {
            ind=-1;
        }
    }
    printf("%d",ind);
    return 0;
}
