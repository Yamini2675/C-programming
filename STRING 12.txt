#include<stdio.h>

int main() {
    char str[100];
    fgets(str,sizeof(str),stdin);
    int count=0;
    for (int i=0;str[i]!='\0';i++) {
        int found=0;
        if (str[i]>='A' && str[i]<='z') {
            found=0;
        }
        else if (str[i]>='0' && str[i]<='9') {
            found=0;
        }
        else if (str[i]=='\n') {
            found=0;
        }
        else {
            found=1;
            count++;
        }

    }
    printf("%d",count);

    return 0;
}
