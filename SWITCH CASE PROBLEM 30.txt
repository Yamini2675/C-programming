#include<stdio.h>

int main() {
    int hours,category;
    scanf("%d",&hours);

    if (hours<=1) {
        category=1;
    }
    else if (hours<=4) {
        category=2;
    }
    else {
        category=3;
    }

    switch (category) {
        case 1:
            printf("Short Interruption");
            break;
        case 2:
            printf("Medium Interruption");
            break;
        case 3:
            printf("Long Interruption");
            break;
    }
    return 0;
}
