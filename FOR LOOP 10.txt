#include<stdio.h>

int main() {
    int n,daily_usage,total=0,count=0;
    scanf("%d",&n);

    for (int i=1;i<=n;i++) {
        scanf("%d",&daily_usage);
        total=total+daily_usage;
        if (daily_usage>2) {
            count++;
        }
    }
    printf("Total Data:%d\n",total);
    printf("High Usage Days:%d",count);
    return 0;
}
