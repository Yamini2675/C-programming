#include<stdio.h>

int main() {
    int del_days,category,fine;
    scanf("%d",&del_days);

    if (del_days<=5) {
        category=1;
    }
    else if (del_days<=10) {
        category=2;
    }
    else {
        category=3;
    }

    switch (category) {
        case 1:
            fine=del_days*50;
            printf("Fine:%d",fine);
            break;
        case 2:
            fine=del_days*100;
            printf("Fine:%d",fine);
            break;
        case 3:
            fine=del_days*200;
            printf("Fine:%d",fine);
            break;
        default:
            printf("Invalid category");
    }
    return 0;
}
