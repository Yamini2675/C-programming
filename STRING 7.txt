#include<stdio.h>

int main() {
    char str[100];
    int count=0;
    scanf("%s",str);
    for (int i=0;str[i]!='\0';i++) {
        int found=0;
        if (str[i]=='a' || str[i]=='e' || str[i]=='i' ||str[i]=='o' ||str[i]=='u'|| str[i]=='A' || str[i]=='E' || str[i]=='I' || str[i]=='O' || str[i]=='U') {
            found=0;
        }
        else if (str[i]>='0' && str[i]<='9') {
            found=0;
        }
        else {
            found=1;
            count++;
        }
    }

    printf("%d",count);
    return 0;
}
