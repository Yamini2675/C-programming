#include<stdio.h>

int main() {
    int call_type,min,charge;
    scanf("%d %d",&call_type,&min);

    switch (call_type) {
        case 1:
            charge=min*1;
            printf("Call Charge:%d",charge);
            break;
        case 2:
            charge=min*3;
            printf("Call Charge:%d",charge);
            break;
        case 3:
            charge=min*10;
            printf("Call Charge:%d",charge);
            break;
        default:
            printf("Invalid Call Type");
    }
    return 0;
}
