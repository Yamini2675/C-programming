#include<stdio.h>

int main() {
    int r,c;
    scanf("%d %d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }

    int max=0;
    int sec_max=0;
    for (int k=0;k<r;k++) {
        for (int h=0;h<c;h++) {
            if (arr[k][h]>max) {
                max=arr[k][h];
            }
            if (arr[k][h]<max && arr[k][h]>sec_max) {
                sec_max=arr[k][h];
            }
        }
    }
    printf("Highest salary=%d\n",max);
    printf("Second highest distinct salary=%d",sec_max);

    return 0;
}
