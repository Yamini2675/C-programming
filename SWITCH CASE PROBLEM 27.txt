#include<stdio.h>

int main() {
    int del_mode;
    scanf("%d",&del_mode);

    switch (del_mode) {
        case 1:
            printf("Delivery Charge:40");
            break;
        case 2:
            printf("Delivery Charge:120");
            break;
        case 3:
            printf("Delivery Charge:0(Free)");
            break;
        default:
            printf("Invalid Mode");
    }
    return 0;
}
