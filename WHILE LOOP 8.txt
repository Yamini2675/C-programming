#include<stdio.h>

int main() {
    int months,n,count=0;
    int max=0,i=1;

    scanf("%d",&months);

    while (i<months) {
        scanf("%d",&n);
        if (n==0) {
            count++;
        }
        else {
            count=0;
        }
        if (count>max) {
            max=count;
        }
        i++;
    }
    printf("Longest Default Streak:%d\n",max);
    return 0;
}
