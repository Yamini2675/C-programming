#include<stdio.h>

void fibo(int n) {
    int first=0,second=1;
    int temp;
    if (1<=n) {
        printf("%d ",first);
    }
    if (2<=n) {
        printf("%d ",second);
    }
    for (int i=2;i<n;i++) {
        temp=first+second;
        first=second;
        second=temp;
        printf("%d ",temp);
    }
}

int main() {
    int n;
    scanf("%d",&n);
    fibo(n);

    return 0;
}
