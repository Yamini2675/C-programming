#include<stdio.h>

int main() {
    int r,c;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int max=0,min=0;
    int ind=0;
    for (int i=0;i<r;i++) {
        int pro=1;
        for (int j=0;j<c;j++) {
            pro=pro*arr[i][j];
        }
        if (pro>max) {
            max=pro;
            ind=i;
        }
        if (pro<max) {
            min=pro;
        }
    }
    printf("%d",ind);

    return 0;
}
