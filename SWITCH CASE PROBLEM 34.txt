#include<stdio.h>

int main() {
    int plan_type;
    scanf("%d",&plan_type);

    switch (plan_type) {
        case 1:
            printf("Speed 40 Mbps");
            break;
        case 2:
            printf("Speed 100 Mbps");
            break;
        case 3:
            printf("Speed 300 Mbps");
            break;
        default:
            printf("Invalid Plan");
    }
    return 0;
}
