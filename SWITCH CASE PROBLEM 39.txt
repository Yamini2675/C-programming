#include<stdio.h>

int main() {
    int dist_category;
    scanf("%d",&dist_category);

    switch (dist_category) {
        case 1:
            printf("Transport Fee:800");
            break;
        case 2:
            printf("Transport Fee:1200");
            break;
        case 3:
            printf("Transport Fee:1800");
            break;
        default:
            printf("Invalid Distance Category");
    }
    return 0;
}
