#include<stdio.h>

int main() {
    int r,c;
    scanf("%d %d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int max=0,ind=0;
    for (int j=0;j<c;j++) {
        int sum=0;
        for (int i=0;i<r;i++) {
                sum=sum+arr[i][j];
        }
        if (sum>max) {
            max=sum;
            ind=j;
        }
    }
    printf("%d",ind);

    return 0;
}
