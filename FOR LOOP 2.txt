#include<stdio.h>

int main() {
    int n,amt,total=0;
    scanf("%d",&n);

    for (int i=1;i<=n;i++) {
        scanf("%d",&amt);
        total=total+amt;
    }
    if (total<=10000) {
        printf("Approved");
    }
    else {
        printf("Limit Exceeded");
    }
    return 0;
}
