#include<stdio.h>

int main() {
    int n,delay_days,count=0;
    int total=0,i=1;

    scanf("%d",&n);

    while (i<=n) {
        scanf("%d",&delay_days);
        total=total+delay_days;
        if (delay_days>2) {
            count++;
        }
        i++;
    }
    printf("Total Delay : %d\n",total);
    printf("Delayed Days : %d",count);

    return 0;
}
