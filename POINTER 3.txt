#include<stdio.h>

int main() {
    char str[100];
    fgets(str,sizeof(str),stdin);
    char *p;
    p=str;
    int count=0;
    while(*p!='\0') {
        if (*p=='a'||*p=='A'||*p=='e'||*p=='E'||*p=='i'||*p=='I'|| *p=='o'||*p=='O'|| *p=='u'||*p=='U') {
            count++;
        }
        p++;
    }
    printf("%d",count);

    return 0;
}
