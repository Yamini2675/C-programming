#include<stdio.h>

int main() {
    int attempt,mark;
    printf("Enter Attempt:");
    scanf("%d",&attempt);
    printf("Enter Mark:");
    scanf("%d",&mark);

    switch (attempt) {
        case 1:
            if (mark>=80) {
                printf("Excellent");
            }
            else {
                printf("Not Qualified");
            }
            break;
        case 2:
            if (mark>=60) {
                printf("Good");
            }
            else {
                printf("Not Qualified");
            }
            break;
        case 3:
            printf("Need Improvement");
            break;
        default:
            printf("Need Improvement");
    }
    return 0;
}
