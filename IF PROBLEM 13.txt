#include<stdio.h>

int main() {
    char ch;
    scanf("%c",&ch);

    if (ch>='A' && ch<='Z') {
        printf("Uppercase Letter");
    }
    else if (ch>='a' && ch<='z') {
        printf("Lowercase Letter");
    }
    else {
        printf("Invalid Character");
    }
    return 0;
}
