#include<stdio.h>

int main() {
    int days,rainfall,total=0,count=0;
    scanf("%d",&days);

    for (int i=1;i<=days;i++) {
        scanf("%d",&rainfall);
        total=total+rainfall;
        if (rainfall>50) {
            count++;
        }
    }
    printf("Total Rainfall:%d\n",total);
    printf("Heavy Rain Days:%d",count);
    return 0;
}
