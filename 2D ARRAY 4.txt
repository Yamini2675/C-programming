#include<stdio.h>

int main() {
    int r,c;
    scanf("%d %d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int found=0;
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {

            for (int k=i;k<r;k++) {
                for (int l=0;l<c;l++) {
                    if (k==i && l<=j) {
                        continue;
                    }
                    if (arr[i][j]==arr[k][l]) {
                        printf("%d",arr[i][j]);
                        found=1;
                        break;
                    }
                }
                if (found) {
                    break;
                }
            }
            if (found) {
                break;
            }
        }
        if (found) {
            break;
        }
    }
    if (!found) {
        printf("No repeating element");
    }

    return 0;
}
