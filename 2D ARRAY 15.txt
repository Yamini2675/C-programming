#include<stdio.h>

int main() {
    int r,c;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int count=0;
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            int fact=1;
            if (arr[i][j]<=1) {
                fact=0;
            }
            else {
                for (int k=2;k<=arr[i][j]/2;k++) {
                    if (arr[i][j]%k==0) {
                        fact=0;
                        break;
                    }
                }
            }
            if (fact==1) {
                count++;
            }
        }
    }
    printf("%d",count);

    return 0;
}
