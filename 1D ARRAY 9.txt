#include<stdio.h>

int main() {
    int n;
    scanf("%d",&n);
    int arr[100];
    int leader[100];

    for(int i=0;i<n;i++) {
        scanf("%d",&arr[i]);
    }
    int max=arr[n-1];
    int count=0;

    leader[count]=max;
    count++;

    for (int i=n-2;i>=0;i--) {
        if (arr[i]>max) {
            max=arr[i];
            leader[count]=max;
            count++;
        }
    }
    for (int i=count-1;i>=0;i--) {
        printf("%d ",leader[i]);
    }
    return 0;
}
