#include<stdio.h>

int main() {
    int r,c;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            scanf("%d",&arr[i][j]);
        }
    }
    int ase=0;
    int count=0;
    for (int i=0;i<r;i++) {
        for (int j=0;j<c;j++) {
            if (arr[i][j]<arr[i][j+1]) {
                ase=1;
            }
            if (arr[i][j]>arr[i][j+1]) {
                ase=0;
                break;
            }
        }
        if (ase==1) {
            count++;
        }
    }
    printf("%d\n",count);

    return 0;
}
