#include<stdio.h>

int main() {
    int n;
    scanf("%d",&n);
    int arr[n];

    for(int i=0;i<n;i++) {
        scanf("%d",&arr[i]);
    }
    int total=0;

    for (int i=0;i<n;i++) {
        total+=arr[i];
    }
    int avg;
    avg=total/n;

    int count=0;
    for (int j=0;j<n;j++) {
        if (arr[j]>avg) {
            count++;
        }
    }
    printf("%d",count);

    return 0;
}
