#include<stdio.h>

int main() {
    int n,missing;
    scanf("%d",&n);
    int arr[n];
    for (int i=0;i<n-1;i++) {
        scanf("%d",&arr[i]);
    }
    int sum=0;
    sum=n*(n+1)/2;
    int total=0;
    for (int i=0;i<n-1;i++) {
        total=total+arr[i];
    }
    missing=sum-total;
    printf("%d",missing);
    return 0;
}
