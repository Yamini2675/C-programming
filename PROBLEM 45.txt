#include<stdio.h>

int main() {
     int temp;
     printf("Enter Temperature:");
     scanf("%d",&temp);

     if (temp<20) {
          printf("Heating Mode");
     }
     else if (temp>=20 && temp<=30) {
          printf("Normal Mode");
     }
     else {
          printf("Cooling Mode");
     }
     return 0;
}
