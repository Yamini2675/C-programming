#include<stdio.h>

int main() {
     int num;
     printf("Enter a Number:");
     scanf("%d",&num);

     if (num==1) {
          printf("Admin");
     }
     else if (num==2) {
          printf("User");
     }
     else if (num==3) {
          printf("Guest");
     }
     else {
          printf("Invalid Number");
     }
     return 0;
}
