#include<stdio.h>

int main() {
    int n,i,max;
    int arr[1000];
    int *p=arr;
    scanf("%d",&n);
    for(i=0;i<n;i++) {
        scanf("%d",&arr[i]);
    }
    max=*p;
    for(i=0;i<n;i++) {
        if (*(p+i)>max) {
            max=*(p+i);
        }
    }
    printf("%d",max);

    return 0;
}
